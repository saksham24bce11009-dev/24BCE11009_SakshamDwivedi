use vitcampus;\
\
-- Parent Table\
create table  table1 (\
    Category_ID int primary key,\
    Category_Name varchar(50) not null\
);\
\
\
-- Child Table\
create table table2 (\
    Product_ID int primary key,\
    Product_Name varchar(50) not null,\
    Category_ID INT,\
    \
    foreign key (Category_ID) references table1(Category_ID)\
);\
\
-- Inserting values into parent table\
insert into table1 values(1, 'Electronics'),(2, 'Clothing'),(3, 'Books');\
\
-- Inserting values into child table\
